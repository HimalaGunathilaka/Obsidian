>Code used to generate `A`
```python
import numpy as np

size = 6
J = np.zeros((size, size))

# Set up the Jordan matrix J
# Diagonals
J[0, 0] = J[1, 1] = J[2, 2] = J[3, 3] = 2
J[4, 4] = J[5, 5] = 2

# Superdiagonals (1s above diagonal for Jordan blocks)
J[0, 1] = J[1, 2] = J[2, 3] = 1
J[4, 5] = 1

# Step 2: Random invertible matrix P using integers
# Assuming the algebraic multiplicities of Jordan blocks are 4 and 2

while True:
    # Generate random integer matrix P
    P = np.random.randint(-1, 2, (6, 6))  # Random integers between -1 and 1

    # Check if P is invertible
    if np.linalg.det(P) == 0:
        continue

    # Compute A = P * J * P⁻¹
    A = P @ J @ np.linalg.inv(P)

    # Check if A is upper triangular
    is_upper = np.allclose(np.tril(A, k=-1), 0, atol=1e-10)

    if is_upper:
        print("Jordan form J:")
        print(J)

        print("\nRandom invertible matrix P:")
        print(P)

        print("\nInverse of P:")
        print(np.linalg.inv(P))

        print("\nFinal matrix A (upper triangular):")
        print(A)
        break
```
